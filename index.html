# CELVRA — Beyond Storage, Beyond Limits

> **Crystal Drive Platform** · Built in 2026 by **Kerolos Adly**

---

## 🔮 What is Celvra?

**Celvra** is a next-generation personal cloud storage platform featuring the revolutionary **Crystal Drive Interface** — a visually breathtaking, animated UI that redefines what personal file management looks like. Designed for complete privacy, unlimited storage, and seamless cross-device access.

---

## ✨ Features

### 🔐 Authentication
- **Secure Registration** — Email address can only be used once across the platform
- **Remember Me** — Save login credentials so you never have to enter them again
- **Cross-Device Sync** — Your data follows you on any device you log in from
- **Persistent Sessions** — Files and settings are saved even after closing the browser

### 📁 File Management
| Section | Supported Formats |
|---------|------------------|
| 📸 Photos | JPG, PNG, GIF, WEBP, SVG, BMP |
| 🎵 Music | MP3, WAV, FLAC, AAC, OGG, M4A, WMA |
| 👤 Contacts | VCF (vCard format only) |
| 🎬 Videos | MP4, MKV, AVI, MOV, WEBM, FLV |
| 📄 Documents | PDF, DOCX, DOC, XLSX, XLS, TXT |
| 📱 APK Files | Android Application Packages |
| 💻 EXE Files | Windows x86 & x64 Executables |
| 🔗 Links | Named bookmarks with custom URLs |

### 🗂️ Folder System
- Create custom folders in any section
- Name folders as you like
- Upload files directly into folders
- Delete folders (and all their contents)
- Files inside folders do not appear outside them

### 📤 File Transfer
- Send any file to another Celvra user using their **email address**
- The sender's email is hidden — only their **username** is shown to the recipient
- Recipient gets a **pop-up notification** + **sound alert**
- Notification appears on any page: *"You received a notification — Visit the Notification Center"*
- Recipient can **Accept & Save** the file directly to their gallery
- Accepted files go to their appropriate section (photo → Photos, etc.)

### 🔔 Notification System
- Real-time notification sound when a file is received
- Toast pop-ups appear on any page of the platform
- Notification center on the Dashboard
- Delete individual notifications
- Unread badge count on Bell icon and Sidebar

### 🎨 Crystal Drive Interface
- Animated floating crystal orbs & particles
- Dynamic grid overlay with aurora effects
- Glassmorphism cards with crystal borders
- Orbitron font for the logo and brand elements
- Smooth page transitions with Framer Motion
- Fully responsive — Desktop, Tablet, and Mobile

### ☁️ Storage
- **Unlimited storage space** — no file size caps
- All files stored securely in localStorage per user
- Data persists across browser sessions

## 🚀 Getting Started

### Prerequisites
- Node.js 18+ 
- npm or yarn

### Installation
```bash
git clone https://github.com/your-username/celvra.git
cd celvra
npm install
```

### Development
```bash
npm run dev
```

### Build for Production
```bash
npm run build
```

The built files will be in the `dist/` folder — ready to deploy to GitHub Pages, Netlify, Vercel, or any static host.

---

## 🌐 Deploying to GitHub Pages

1. Build the project: `npm run build`
2. Push the `dist/` folder to your `gh-pages` branch, or use the GitHub Actions workflow
3. In your repository settings, set GitHub Pages source to the `dist/` folder

function TopBar({ onMenuToggle }: { onMenuToggle: () => void }) {
  const { currentUser, getUserNotifications, setActiveSection, activeSection } = useStore();
  const notifications = getUserNotifications();
  const unread = notifications.filter(n => !n.read).length;

  const sectionTitle: Record<string, string> = {
    dashboard: 'Dashboard',
    photos: 'Photos',
    music: 'Music',
    contacts: 'Contacts',
    videos: 'Videos',
    documents: 'Documents',
    apk: 'APK Files',
    exe: 'EXE Files',
    links: 'Links',
    about: 'About Celvra',
  };

  return (
    <div className="sticky top-0 z-30 flex items-center justify-between px-6 py-4 border-b" style={{ background: 'rgba(6,11,24,0.9)', borderColor: 'rgba(99,179,237,0.08)', backdropFilter: 'blur(20px)' }}>
      <div className="flex items-center gap-4">
        <button onClick={onMenuToggle} className="lg:hidden w-9 h-9 rounded-xl flex items-center justify-center" style={{ background: 'rgba(255,255,255,0.05)', color: '#64748b', border: '1px solid rgba(255,255,255,0.08)' }}>
          <Menu className="w-5 h-5" />
        </button>
        <div>
          <h1 className="text-lg font-bold text-white">{sectionTitle[activeSection] || 'Celvra'}</h1>
          <p className="text-xs hidden sm:block" style={{ color: 'rgba(148,163,184,0.5)' }}>Crystal Drive Platform</p>
        </div>
      </div>
      <div className="flex items-center gap-3">
        {/* Notification bell */}
        <button
          onClick={() => setActiveSection('dashboard')}
          className="relative w-9 h-9 rounded-xl flex items-center justify-center transition-all"
          style={{ background: 'rgba(255,255,255,0.04)', border: '1px solid rgba(255,255,255,0.08)', color: '#64748b' }}>
          <Bell className="w-4 h-4" />
          {unread > 0 && (
            <motion.span
              initial={{ scale: 0 }}
              animate={{ scale: 1 }}
              className="absolute -top-1 -right-1 w-4 h-4 rounded-full text-[10px] font-bold flex items-center justify-center"
              style={{ background: 'linear-gradient(135deg, #ef4444, #dc2626)', color: 'white' }}>
              {unread > 9 ? '9+' : unread}
            </motion.span>
          )}
        </button>
        {/* User avatar */}
        {currentUser && (
          <div className="w-9 h-9 rounded-xl flex items-center justify-center font-bold text-sm" style={{ background: 'linear-gradient(135deg, #63b3ed, #8b5cf6)', color: 'white' }}>
            {currentUser.username.charAt(0).toUpperCase()}
          </div>
        )}
      </div>
    </div>
  );
}

export default function App() {
  const { currentUser, activeSection } = useStore();
  const [mobileMenuOpen, setMobileMenuOpen] = useState(false);

  useEffect(() => {
    // The persisted currentUser is loaded automatically from localStorage on init
  }, []);

  if (!currentUser) {
    return <AuthPage />;
  }

  const renderSection = () => {
    if (activeSection === 'dashboard') return <Dashboard />;
    if (activeSection === 'about') return <AboutPage />;
    if (VALID_CATEGORIES.includes(activeSection as FileCategory)) {
      return <FileSection category={activeSection as FileCategory} />;
    }
    return <Dashboard />;
  };

  return (
    <div className="min-h-screen flex relative">
      <CrystalBackground />

      <Sidebar mobileOpen={mobileMenuOpen} onMobileClose={() => setMobileMenuOpen(false)} />

      <main className="flex-1 flex flex-col min-w-0 relative z-10">
        <TopBar onMenuToggle={() => setMobileMenuOpen(true)} />
        <div className="flex-1 overflow-y-auto">
          <div className="max-w-7xl mx-auto p-6">
            <AnimatePresence mode="wait">
              <motion.div
                key={activeSection}
                initial={{ opacity: 0, y: 16 }}
                animate={{ opacity: 1, y: 0 }}
                exit={{ opacity: 0, y: -16 }}
                transition={{ duration: 0.3, ease: 'easeOut' }}
              >
                {renderSection()}
              </motion.div>
            </AnimatePresence>
          </div>
        </div>
      </main>

      {/* Global notification toasts */}
      <NotificationToast />

      {/* Crystal shimmer watermark */}
      <div className="fixed bottom-3 left-1/2 -translate-x-1/2 z-20 pointer-events-none">
        <div className="flex items-center gap-2 opacity-20">
          <Sparkles className="w-3 h-3 text-cyan-400" />
          <span className="text-xs tracking-widest" style={{ fontFamily: 'Orbitron', color: '#63b3ed', fontSize: '0.6rem' }}>CELVRA · CRYSTAL DRIVE</span>
          <Sparkles className="w-3 h-3 text-cyan-400" />
        </div>
      </div>
    </div>
  );
}

function timeAgo(dateStr: string) {
  const diff = Date.now() - new Date(dateStr).getTime();
  const m = Math.floor(diff / 60000);
  if (m < 1) return 'just now';
  if (m < 60) return `${m}m ago`;
  const h = Math.floor(m / 60);
  if (h < 24) return `${h}h ago`;
  return `${Math.floor(h / 24)}d ago`;
}

function DropzoneArea({ category, folderId }: { category: FileCategory; folderId?: string }) {

        addFile({ name: file.name, size: file.size, type: file.type, category, folderId, dataUrl });

import { useEffect, useRef, useState } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { Bell, X, ChevronRight } from 'lucide-react';
import { useStore } from '../store/useStore';

export default function NotificationToast() {
  const { getUserNotifications, currentUser, setActiveSection } = useStore();
  const [toasts, setToasts] = useState<Array<{ id: string; fromUsername: string; fileName: string }>>([]);
  const prevCountRef = useRef(0);
  const audioRef = useRef<HTMLAudioElement | null>(null);

  // Create notification sound using Web Audio API
  const playNotificationSound = () => {
    try {
      const ctx = new (window.AudioContext || (window as unknown as { webkitAudioContext: typeof AudioContext }).webkitAudioContext)();
      const oscillator = ctx.createOscillator();
      const gainNode = ctx.createGain();
      oscillator.connect(gainNode);
      gainNode.connect(ctx.destination);
      oscillator.type = 'sine';
      oscillator.frequency.setValueAtTime(880, ctx.currentTime);
      oscillator.frequency.exponentialRampToValueAtTime(440, ctx.currentTime + 0.3);
      gainNode.gain.setValueAtTime(0.3, ctx.currentTime);
      gainNode.gain.exponentialRampToValueAtTime(0.001, ctx.currentTime + 0.4);
      oscillator.start(ctx.currentTime);
      oscillator.stop(ctx.currentTime + 0.4);
    } catch {}
  };

  useEffect(() => {
    if (!currentUser) return;
    const notifications = getUserNotifications();
    const unreadCount = notifications.filter(n => !n.read).length;
    
    if (unreadCount > prevCountRef.current) {
      const newNotifs = notifications.filter(n => !n.read).slice(0, unreadCount - prevCountRef.current);
      newNotifs.forEach(n => {
        setToasts(prev => [...prev, { id: n.id, fromUsername: n.fromUsername, fileName: n.fileName }]);
        playNotificationSound();
        setTimeout(() => {
          setToasts(prev => prev.filter(t => t.id !== n.id));
        }, 6000);
      });
    }
    prevCountRef.current = unreadCount;
  }, [getUserNotifications, currentUser]);

  // Poll for new notifications every 2 seconds
  useEffect(() => {
    if (!currentUser) return;
    const interval = setInterval(() => {
      const notifications = getUserNotifications();
      const unreadCount = notifications.filter(n => !n.read).length;
      if (unreadCount > prevCountRef.current) {
        const diff = unreadCount - prevCountRef.current;
        const newNotifs = notifications.filter(n => !n.read).slice(0, diff);
        newNotifs.forEach(n => {
          const exists = toasts.find(t => t.id === n.id);
          if (!exists) {
            setToasts(prev => {
              const alreadyThere = prev.find(t => t.id === n.id);
              if (alreadyThere) return prev;
              return [...prev, { id: n.id, fromUsername: n.fromUsername, fileName: n.fileName }];
            });
            playNotificationSound();
            setTimeout(() => {
              setToasts(prev => prev.filter(t => t.id !== n.id));
            }, 6000);
          }
        });
        prevCountRef.current = unreadCount;
      }
    }, 2000);
    return () => clearInterval(interval);
  }, [currentUser, getUserNotifications, toasts]);

  const dismissToast = (id: string) => setToasts(prev => prev.filter(t => t.id !== id));

  return (
    <div className="fixed bottom-6 right-6 z-[100] space-y-3 pointer-events-none">
      <AnimatePresence>
        {toasts.map(toast => (
          <motion.div
            key={toast.id}
            initial={{ opacity: 0, x: 100, scale: 0.8 }}
            animate={{ opacity: 1, x: 0, scale: 1 }}
            exit={{ opacity: 0, x: 100, scale: 0.8 }}
            className="pointer-events-auto flex items-start gap-3 p-4 rounded-2xl max-w-xs w-full shadow-2xl"
            style={{ background: 'rgba(10,15,30,0.97)', border: '1px solid rgba(99,179,237,0.4)', boxShadow: '0 0 40px rgba(99,179,237,0.2)', backdropFilter: 'blur(20px)' }}
          >
            <div className="absolute top-0 left-0 right-0 h-px rounded-t-2xl" style={{ background: 'linear-gradient(90deg, transparent, rgba(99,179,237,0.8), transparent)' }} />
            <div className="w-9 h-9 rounded-xl flex items-center justify-center flex-shrink-0" style={{ background: 'rgba(99,179,237,0.15)', border: '1px solid rgba(99,179,237,0.3)' }}>
              <Bell className="w-4 h-4 text-cyan-400" />
            </div>
            <div className="flex-1 min-w-0">
              <p className="text-white text-sm font-semibold">You received a file!</p>
              <p className="text-xs mt-0.5 truncate" style={{ color: 'rgba(148,163,184,0.7)' }}>
                From <span style={{ color: '#63b3ed' }}>{toast.fromUsername}</span>
              </p>
              <p className="text-xs mt-0.5 truncate" style={{ color: 'rgba(148,163,184,0.5)' }}>{toast.fileName}</p>
              <button
                onClick={() => { setActiveSection('dashboard'); dismissToast(toast.id); }}
                className="flex items-center gap-1 text-xs mt-2 font-medium"
                style={{ color: '#63b3ed' }}>
                Visit Notification Center <ChevronRight className="w-3 h-3" />
              </button>

const loadState = () => {
  try {
    const raw = localStorage.getItem(STORAGE_KEY);
    if (!raw) return null;
    return JSON.parse(raw);
  } catch {
    return null;
  }

const saveState = (state: Partial<CelvraState>) => {
  try {
    const toSave = {
      users: state.users,
      files: state.files,
      folders: state.folders,
      links: state.links,
      notifications: state.notifications,
      currentUser: state.currentUser,
      rememberMe: state.rememberMe,

const persisted = loadState();

export const useStore = create<CelvraState>((set, get) => ({
  users: persisted?.users || [],
  currentUser: persisted?.currentUser || null,
  rememberMe: persisted?.rememberMe || false,
  files: persisted?.files || [],
  folders: persisted?.folders || [],
  links: persisted?.links || [],
  notifications: persisted?.notifications || [],
  activeSection: 'dashboard',
  activeFolderId: null,
  newNotificationSound: false,

  register: (username, email, password) => {
    const state = get();
    if (state.users.find(u => u.email.toLowerCase() === email.toLowerCase())) {
      return { success: false, error: 'This email is already registered.' };
    }
    if (state.users.find(u => u.username.toLowerCase() === username.toLowerCase())) {
      return { success: false, error: 'This username is already taken.' };
    }
    const newUser: User = { id: uuidv4(), username, email: email.toLowerCase(), password, createdAt: new Date().toISOString() };
    const users = [...state.users, newUser];
    set({ users });
    saveState({ ...get(), users });
    return { success: true };
  },

  login: (email, password, remember) => {
    const state = get();
    const user = state.users.find(u => u.email.toLowerCase() === email.toLowerCase() && u.password === password);
    if (!user) return { success: false, error: 'Invalid email or password.' };
    set({ currentUser: user, rememberMe: remember });
    saveState({ ...get(), currentUser: remember ? user : null, rememberMe: remember });
    return { success: true };
  },

  logout: () => {
    const state = get();
    if (!state.rememberMe) {
      set({ currentUser: null });
      saveState({ ...get(), currentUser: null });
    } else {
      set({ currentUser: null });
      saveState({ ...get(), currentUser: null, rememberMe: false });
    }
  },

  checkRemembered: () => {
    const state = get();
    if (state.rememberMe && state.currentUser) {
      // already logged in from persisted state
    }
  },

  addFile: (file) => {
    const state = get();
    if (!state.currentUser) return;
    const newFile: CelvraFile = {
      ...file,
      id: uuidv4(),
      uploadedAt: new Date().toISOString(),
      ownerId: state.currentUser.id,
    };
    const files = [...state.files, newFile];
    set({ files });
    saveState({ ...get(), files });
  },

  deleteFile: (id) => {
    const files = get().files.filter(f => f.id !== id);
    set({ files });
    saveState({ ...get(), files });
  },

  getUserFiles: (category?, folderId?) => {
    const state = get();
    if (!state.currentUser) return [];
    return state.files.filter(f => {
      if (f.ownerId !== state.currentUser!.id) return false;
      if (category && f.category !== category) return false;
      if (folderId !== undefined) {
        if (folderId === null) return !f.folderId;
        return f.folderId === folderId;
      }
      return true;
    });
  },

  addFolder: (name, category) => {
    const state = get();
    if (!state.currentUser) return;
    const folder: CelvraFolder = {
      id: uuidv4(),
      name,
      category,
      ownerId: state.currentUser.id,
      createdAt: new Date().toISOString(),
    };
    const folders = [...state.folders, folder];
    set({ folders });
    saveState({ ...get(), folders });
  },

  deleteFolder: (id) => {
    const state = get();
    // Also delete files inside the folder
    const files = state.files.filter(f => f.folderId !== id);
    const folders = state.folders.filter(f => f.id !== id);
    set({ folders, files });
    saveState({ ...get(), folders, files });
  },

  getUserFolders: (category) => {
    const state = get();
    if (!state.currentUser) return [];
    return state.folders.filter(f => f.ownerId === state.currentUser!.id && f.category === category);
  },

  addLink: (name, url) => {
    const state = get();
    if (!state.currentUser) return;
    const link: CelvraLink = {
      id: uuidv4(),
      name,
      url,
      ownerId: state.currentUser.id,
      createdAt: new Date().toISOString(),
    };
    const links = [...state.links, link];
    set({ links });
    saveState({ ...get(), links });
  },

  deleteLink: (id) => {
    const links = get().links.filter(l => l.id !== id);
    set({ links });
    saveState({ ...get(), links });
  },

  getUserLinks: () => {
    const state = get();
    if (!state.currentUser) return [];
    return state.links.filter(l => l.ownerId === state.currentUser!.id);
  },

  sendFile: (toEmail, file) => {
    const state = get();
    if (!state.currentUser) return { success: false, error: 'Not logged in.' };
    const recipient = state.users.find(u => u.email.toLowerCase() === toEmail.toLowerCase());
    if (!recipient) return { success: false, error: 'No user found with that email.' };
    if (recipient.id === state.currentUser.id) return { success: false, error: 'You cannot send files to yourself.' };
    const notif: Notification = {
      id: uuidv4(),
      toUserId: recipient.id,
      fromUsername: state.currentUser.username,
      fileId: file.id,
      fileName: file.name,
      fileCategory: file.category,
      fileDataUrl: file.dataUrl,
      fileType: file.type,
      fileSize: file.size,
      read: false,
      createdAt: new Date().toISOString(),
    };
    const notifications = [...state.notifications, notif];
    set({ notifications });
    saveState({ ...get(), notifications });
    return { success: true };
  },

  markNotificationRead: (id) => {
    const state = get();
    const notifications = state.notifications.map(n => n.id === id ? { ...n, read: true } : n);
    // Also add the file to the recipient's files
    const notif = state.notifications.find(n => n.id === id);
    if (notif && state.currentUser && !notif.read) {
      const newFile: CelvraFile = {
        id: uuidv4(),
        name: notif.fileName,
        size: notif.fileSize,
        type: notif.fileType,
        category: notif.fileCategory,
        dataUrl: notif.fileDataUrl,
        uploadedAt: new Date().toISOString(),
        ownerId: state.currentUser.id,
      };
      const files = [...state.files, newFile];
      set({ notifications, files });
      saveState({ ...get(), notifications, files });
    } else {
      set({ notifications });
      saveState({ ...get(), notifications });
    }
  },

  deleteNotification: (id) => {
    const notifications = get().notifications.filter(n => n.id !== id);
    set({ notifications });
    saveState({ ...get(), notifications });
  },

  getUserNotifications: () => {
    const state = get();
    if (!state.currentUser) return [];
    return state.notifications.filter(n => n.toUserId === state.currentUser!.id).sort((a, b) => new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime());
  },

  clearNewNotificationSound: () => set({ newNotificationSound: false }),

  setActiveSection: (section) => set({ activeSection: section, activeFolderId: null }),
  setActiveFolderId: (id) => set({ activeFolderId: id }),
}));
